#include<stdio.h>
#define MAX 10
int front=-1,rear=-1;
int queue[MAX];
void insert();
void delete_ele();
void display();

void main()
{
    int op;
    do{
        printf("\n 1.Insert:");
        printf("\n 2.Delete:");
        printf("\n 3.Display:");
        printf("\n 4.Exit:");
        printf("\n Enter Your Choice :");
        scanf("%d",&op);

        switch(op)
        {
            case 1:
                insert();
                break;
            case 2:
                delete_ele();
                break;
            case 3:
                display();
                break;
            case 4:
                printf("\Exit");
                break;
            default:
                printf("\n Invalid Choice");
        }

    }while(op!=4);

}

void insert()
{
    int num;
    printf("\n Enter value of inserted :");
    scanf("%d",&num);

    if(rear==MAX-1)
    {
        printf("\n Queue is Overflow");
        return;
    }
    else if(front==-1 && rear==-1)
    {
        front=rear=0;
    }
    else
    {
    rear++;

    }
queue[rear] = num;
}
void delete_ele()
{
    int val;
    if(front==-1)
    {
        printf("\n Queue is Underflow");
        return;
    }
    else
    {
        val=queue[front];
        printf("\n Value is Deleted is: %d",val);

        if(front==rear)
        {
            front=rear=-1;
        }
        else
        {
            front++;
        }

    }

}
void display()
{
    int i;
    if(front==-1)
    {
        printf("\n Queue is Empty");
        return;
    }
    else
    {
        for(i=front;i<=rear;i++)
        {
            printf("\t %d",queue[i]);
        }
    }
}
